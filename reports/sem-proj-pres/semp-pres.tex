\begin{frame}
  \frametitle{What is a FlowPool}
  \framesubtitle{Big Picture}

  \begin{block}{Collection}
    \begin{itemize}
    \item Deterministic
    \item Lock-Free
    \item Asynchronous
    \item Unordered
    \end{itemize}
  \end{block}

  \pause

  \begin{block}{Determinism}
    Every execution of a given program with given input eventually
    \begin{itemize}
    \item Always reaches the same state
    \item[] \qquad or
    \item Always fails
    \end{itemize}
  \end{block}

\end{frame}
